# Шаг 3: Lua Логика и Сеть

Чтобы вентилятор "ожил", нужно написать для него класс, унаследованный от базового `KineticBlock`, и зарегистрировать события движка.

## 1. Модуль Механизма (Класс)

Создайте файл `modules/kinetics/MechanicalFan.lua`. В нём мы зададим параметры потребителя (нагрузку) и логику отрисовки.

```lua
require "utils"
local KineticBlock = require "KineticBlock"
local Registry = require "network/KineticRegistry"

---@class MechanicalFan : KineticBlock
local MechanicalFan = setmetatable({}, {__index = KineticBlock})
MechanicalFan.__index = MechanicalFan

-- Настройки потребителя
MechanicalFan.is_generator = false
MechanicalFan.base_impact = 4 -- Вентилятор потребляет 4 единицы нагрузки при RPM = 1, если скорость сети 32, то потребление будет 32*4

function MechanicalFan:on_placed()
    KineticBlock.on_placed(self)
    
    -- Спавним сущность ровно по центру блока
    local ent = entities.spawn(make_entity_prefix("mechanical_fan"), make_position_offs(self.x, self.y, self.z, 0.5))

    if ent then
        ent.transform:set_rot(self:get_base_rotation())
        self:set_entity_uid(ent:get_uid())
        self:sync_entity_uid(true)
    end

    -- Подключаемся к графу сети
    self:apply_network()
end

function MechanicalFan:on_render(gtime, dt)
    KineticBlock.on_render(self, gtime, dt)

    local uid = self.entity_uid
    if uid == -1 or not entities.exists(uid) then return end

    local ent = entities.get(uid)
    local dir = self:get_direction()
    local rpm = self:get_rpm()

    -- Расчет текущего угла поворота на основе RPM
    local current_angle = (gtime * rpm * 6 * dir) % 360

    -- Вращаем саму сущность (если блок поставили боком)
    local base_mat = self:get_base_rotation()
    ent.transform:set_rot(base_mat)

    -- Вращаем конкретно кость лопастей
    local rig = ent.skeleton
    local bone_idx = rig:index("rotor")
    
    if bone_idx then
        -- Вращаем по оси Z (ось вала вентилятора)
        local spin_mat = mat4.rotate({0, 0, 1}, current_angle)
        rig:set_matrix(bone_idx, spin_mat)
    end
end

return MechanicalFan

```

## 2. Скрипт Блока (Связь с VoxelCore)

Движку нужен входной скрипт, чтобы перенаправлять события в наш ООП-класс.
Создайте файл `scripts/mechanical_fan.lua`:

```lua
require "utils"
local Fan = require "kinetics/MechanicalFan"
local Registry = require "network/KineticRegistry"

function on_placed(x, y, z, playerid)
    smart_placement(x, y, z, playerid) -- Умный поворот по взгляду игрока
    local fan = Fan:new(x, y, z)
    fan:on_placed()
end

function on_broken(x, y, z, playerid)
    local fan = Registry.get(x, y, z)
    if fan then fan:on_broken() end
end

-- Восстановление сети при загрузке чанка
function on_block_present(x, y, z)
    KineticBlock.handle_present(Fan, x, y, z)
end

-- Очистка кэша при выгрузке
function on_block_removed(x, y, z)
    KineticBlock.handle_removed(x, y, z)
end

```
